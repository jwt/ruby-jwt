def errors(context, *options)
  errors = []
  iterate_verifiers(*options) do |verifier, verifier_options|
    verify_one!(context, verifier, verifier_options)
  rescue ::JWT::DecodeError => e
    errors << Error.new(message: e.message)
  end
  errors
end
