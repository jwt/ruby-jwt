def kid_generator_type=(value)
  self.kid_generator = case value
                       when :key_digest
                         JWT::JWK::KidAsKeyDigest
                       when :rfc7638_thumbprint
                         JWT::JWK::Thumbprint
                       else
                         raise ArgumentError, "#{value} is not a valid kid generator type."
                       end
end
